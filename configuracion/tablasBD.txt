
--tabla usuarios--
CREATE TABLE usuarios (
    id INT NOT NULL AUTO_INCREMENT,
    nombre VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL,
    contraseña VARCHAR(255) NOT NULL,
    rol ENUM('administrador', 'cliente') DEFAULT 'cliente',
    fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id)
);


----TABLA RESERVAS CORRECTA----
CREATE TABLE reservas (
    id_reserva INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    id_usuario INT NOT NULL,
    nombre_Reservante VARCHAR(100) NOT NULL,
    fechaEntrada DATE NOT NULL,
    fechaSalida DATE NOT NULL,
    adultos INT NOT NULL,
    niños INT NOT NULL,
    habitacion INT NOT NULL,
    email VARCHAR(100) NOT NULL,
    FOREIGN KEY (id_usuario) REFERENCES usuarios (id) 
);

--tabla viaje--
CREATE TABLE viaje(
    id_viaje INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    cantidad INT NOT NULL,
    destino VARCHAR(100),
    vehiculo VARCHAR(50) NOT NULL,
    conductor_nombre VARCHAR(50),
    conductor_id INT NOT NULL,
    FOREIGN KEY (conductor_id) REFERENCES conductor(id_conductor)
);

--tabla conductor--
CREATE TABLE conductor(
    id_conductor INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    nombre_conductor VARCHAR(50),
    vehiculo VARCHAR(50),
    contacto INT NOT NULL
);
